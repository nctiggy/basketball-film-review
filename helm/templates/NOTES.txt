1. Wait for all pods to be ready:

   kubectl get pods -n {{ .Release.Namespace }} -w

2. Get the application URL:

{{- if .Values.ingress.enabled }}
{{- range $host := .Values.ingress.hosts }}
   http{{ if $.Values.ingress.tls }}s{{ end }}://{{ $host.host }}
{{- end }}
{{- else if eq .Values.frontend.service.type "LoadBalancer" }}

   NOTE: It may take a few minutes for the LoadBalancer IP to be available.
   
   export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ .Values.app.name }}-frontend -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
   echo "Access the application at: http://$SERVICE_IP:{{ .Values.frontend.service.port }}"

{{- else if eq .Values.frontend.service.type "NodePort" }}

   export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ .Values.app.name }}-frontend)
   export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
   echo "Access the application at: http://$NODE_IP:$NODE_PORT"

{{- else }}

   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ .Values.app.name }}-frontend 8080:{{ .Values.frontend.service.port }}
   echo "Access the application at: http://localhost:8080"

{{- end }}

3. Access MinIO Console (optional):

   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ .Release.Name }}-minio 9001:9001
   
   Then open: http://localhost:9001
   Username: {{ .Values.minio.rootUser }}
   Password: [check your secrets]

4. Monitor the application:

   # Watch pods
   kubectl get pods -n {{ .Release.Namespace }} -w
   
   # View backend logs
   kubectl logs -n {{ .Release.Namespace }} -l component=backend -f
   
   # View frontend logs
   kubectl logs -n {{ .Release.Namespace }} -l component=frontend -f

5. Database access (if needed):

   kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ .Release.Name }}-postgresql 5432:5432
   
   Then connect with:
   psql -h localhost -U {{ .Values.postgresql.auth.username }} -d {{ .Values.postgresql.auth.database }}
